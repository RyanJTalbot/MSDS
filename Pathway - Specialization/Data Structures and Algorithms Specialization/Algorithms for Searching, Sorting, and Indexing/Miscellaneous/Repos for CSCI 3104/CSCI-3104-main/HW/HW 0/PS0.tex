\begin{verbatim}
\begin{align*}
  16 &= 4 \cdot 4      \\
     &= 2^2 \cdot 2^2  \\
     &= 2^4 .
\end{align*}
\end{verbatim}
This produces:
\begin{align*}
  16 &= 4 \cdot 4      \\
     &= 2^2 \cdot 2^2  \\
     &= 2^4 .
